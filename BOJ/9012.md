## 5. 최종 코드

```cpp
#include <iostream>
#include <string>
using namespace std;

int main (void) {
    int T = 0;
    cin >> T;
    for(int i = 0; i < T; i++) {
        string testcase;
        cin >> testcase;
        int open = 0, close = 0;
        int flag = 1;
        for(int j = 0; j < testcase.length(); j++) {
           if(testcase[j] == '(') open++;
           if(testcase[j] == ')') close++;
           if(open < close) {
            flag = 0;
            break;
           }
        }
        cout << ((flag&&open==close)?"YES":"NO") << "\n";
    }
}
```
