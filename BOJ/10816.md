## 2. 코드 설계 (Pseudo Code)

```cpp
#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int main (void) {
    int N = 0, M = 0;
    cin >> N;
    vector<int> nums(N);
    for(int i = 0; i < N; i++) {
        cin >> nums[i];
    }
    sort(nums.begin(), nums.end());
    cin >> M;
    vector<int> res(M);
    for(int i = 0; i < M; i++) {
        int foo = 0, ans = 0;
        cin >> foo;
        for(int j = 0; j < N; j++) {
            if(nums[j]<foo) continue;
            if(nums[j]>foo) break;
            if(nums[j]==foo) ans++;
        }
        res[i] = ans;
    }
    for(int i = 0; i < M; i++) {
        cout << res[i] << ' ';
        if (i == M-1) cout << '\n';
    }
    return 0;
}
```
