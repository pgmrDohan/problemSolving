#include <iostream>
#include <queue>
#include <utility>
using namespace std;

// 이거 말고 우선순위 큐랑 일반 큐랑 만들어서 우선순위 큐의 top과 비교하는게 더 빠름
bool isMaxPriority(queue<pair<int,int>>& q, int p) {
    int size = q.size();
    int ans = 1;
    for(int i = 0; i<size; i++) {
        const auto temp = q.front();
        q.pop();
        if(temp.first > p) {
            ans = 0;
        }
        q.push(temp);
    }
    return ans;
}

int main (void) {
    ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);
    int N = 0;
    cin >> N;
    for(int i = 0; i<N; i++) {
        int ans = 0;
        int docsN = 0, quriousI = 0;
        cin >> docsN >> quriousI;
        queue<pair<int,int>> q;
        for(int j = 0; j<docsN; j++) {
            int foo = 0;
            cin >> foo;
            q.push({foo,j});
        }
        while(1) {
            const auto temp = q.front();
            q.pop();
            if(isMaxPriority(q,temp.first)) {
                ans++;
                if(temp.second == quriousI) {
                    cout << ans << '\n';
                    break;
                }
            } else {
                q.push(temp);
            }
        }
    }
    return 0;
}
