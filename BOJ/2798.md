## 5. 최종 코드

```cpp
#include <iostream>
#include <vector>

using namespace std;

int main (void) {
    int N, M = 0;
    cin >> N >> M;
    vector<int> nums(N);
    int ans = 0;
    for(int i = 0; i<N; i++) {
        cin >> nums[i];
    }
    for(int i = 0; i<N-2; i++) {
        for(int j = 0; j<N-i-1; j++) {
            for(int k = 0; k<N-i-j-2; k++) {
                int foo = nums[i]+nums[i+j+1]+nums[i+j+k+2];
                if(foo > ans && foo <= M) ans = foo;
            }
        }
    }
    cout << ans << "\n";
    return 0;
}
```
