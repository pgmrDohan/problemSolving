## 5. 최종 코드

```cpp
#include <iostream>
#include <algorithm>
#include <utility>
#include <vector>
using namespace std;

int main (void) {
    ios::sync_with_stdio(false);cin.tie(0);cout.tie(0);
    int N = 0;
    cin >> N;
    vector<pair<int,int>> ps(N);
    for(int i = 0; i<N; i++) {
        cin >> ps[i].first >> ps[i].second;
    }
    sort(ps.begin(),ps.end());
    for(int i = 0; i<N; i++) {
        cout << ps[i].first << ' ' << ps[i].second << '\n';
    }
    return 0;
}
```
