## 5. 최종 코드

```cpp
#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
using namespace std;

int main (void) {
    ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    int N = 0;
    int sum  = 0;
    cin >> N;
    vector<int> v(N);
    for(int i = 0; i<N; i++) {
        cin >> v[i];
        sum += v[i];
    }
    cout << (int)round((double)sum / N) << '\n';
    sort(v.begin(), v.end());
    cout << v[N/2] << '\n';
    int c = 1, maxc = 1;
    int ans = v[0];
    bool second = false;

    for(int i = 1; i < N; i++) {
        if(v[i] == v[i-1]) c++;
        else c = 1;

        if(c > maxc) {
            maxc = c;
            ans = v[i];
            second = false;
        }
        else if(c == maxc && !second) {
            ans = v[i];
            second = true;
        }
    }
    cout << ans << '\n';
    cout << v[N-1] - v[0] << '\n';
    return 0;
}
```
