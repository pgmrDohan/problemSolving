## 5. 최종 코드

```cpp
#include <iostream>
#include <stack>
using namespace std;

int main (void) {
    int N = 0;
    cin >> N;
    stack<int> s;
    for(int i = 0; i<N; i++) {
        int foo = 0;
        cin >> foo;
        if(foo==0) {s.pop();}
        else s.push(foo);
    }
    int sum = 0;
    int siz =s.size();
    for(int i = 0; i<siz; i++) {
        sum += s.top();
        s.pop();
    }
    cout << sum << '\n';
    return 0;
}
```
