## 5. 최종 코드

```cpp
#include <iostream>
#include <string>
#include <stack>
using namespace std;

int main (void) {
    string target = "";
    while(target != ".") {
        getline(cin, target);
        if(target == ".") break;
        stack<char> s;
        int flag = 1;
        for(int i = 0; i < target.length(); i++) {
            if(target[i] == '(') s.push(target[i]);
            if(target[i] == '[') s.push(target[i]);
            if(target[i] == ')') {
                if(s.empty()) {
                    flag = 0;
                    break;
                }
                if(s.top() == '(') {
                    s.pop();
                } else {
                    flag = 0;
                    break;
                }
            }
            if(target[i] == ']') {
                if(s.empty()) {
                    flag = 0;
                    break;
                }
                if(s.top() == '[') {
                    s.pop();
                } else {
                    flag = 0;
                    break;
                }
            }
        }
        if(flag && s.empty()) {
            cout << "yes" << '\n';
        } else {
            cout << "no" << '\n';
        }
    }
    return 0;
}
```
