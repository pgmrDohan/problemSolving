## 5. 최종 코드

```cpp
#include <iostream>
using namespace std;

bool isPrime (int i) {
    if(i<2) return 0;
    for(int j = 2; j*j<=i; j++) {
        if(i%j==0 && j!=i) return 0;
    }
    return 1;
}

int main (void) {
    ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);
    int N = 0, M = 0;
    cin >> N >> M;
    for(int i = N; i <= M; i++) {
        if(isPrime(i)) cout << i << '\n';
    }
    return 0;
}
```
