## 5. 최종 코드

```cpp
#include <iostream>
#include <array>
using namespace std;

int main (void) {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int N = 0;
    cin >> N;
    array<int,10000> count{};
    for(int i = 0; i<N; i++) {
        int foo = 0;
        cin >> foo;
        count[foo-1]++;
    }
    for(int i = 0; i<10000; i++) {
        if(count[i] == 0) continue;
        for(int j = 0; j<count[i]; j++) {
            cout << i+1 << '\n';
        }
    }
    return 0;
}
```
