## 5. 최종 코드

```cpp
#include <iostream>
#include <queue>
using namespace std;

int main (void) {
    queue<int> nums;
    int foo = 0;
    cin >> foo;
    for(int i = 1; i<=foo; i++) {
        nums.push(i);
    }
    while(nums.size() != 1) {
        nums.pop();
        int bar = nums.front();
        nums.pop();
        nums.push(bar);
    }
    cout << nums.front() << '\n';
    return 0;
}
```
