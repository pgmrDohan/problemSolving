## 5. 최종 코드

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
using namespace std;

int main (void) {
    int N = 0;
    cin >> N;
    if(!N) {
        cout << 0 << '\n';
        return 0;
    }
    vector<int> v(N);
    for(int i = 0; i < N; i++) {
        cin >> v[i];
    }
    sort(v.begin(), v.end());
    int cur = (int)round((double)N*3/20);
    int sum = 0;
    for(int i = cur; i < N-cur; i++) {
        sum += v[i];
    }
    cout << (int)round((double)sum/((double)N-2*cur)) << '\n';
    return 0;
}
```
