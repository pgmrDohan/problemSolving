#include <iostream>
#include <vector>
using namespace std;

int static_add (int range, int a, int b) {
    return a+b<range?a+b:(a+b)%range;
}

int main (void) {
    int N = 0, id = 0, foo = 0;
    cin >> N >> id;
    vector<int> nums(N);
    for(int i = 0; i<N; i++) nums[i]=i+1;
    cout << '<';
    for(int i = 0; i<N; i++) {
        //cout << foo;
        foo = static_add(N-i,foo,id-1);
        //cout << N-i << foo << id-1;
        cout << nums[foo];
        for(int j = foo; j+1<N; j++) nums[j]=nums[j+1];
        //for(int i = 0; i<N; i++) cout << nums[i];
        if(i!=N-1) cout <<  ", ";
    }
    cout << '>' << '\n';
    return 0;
}
