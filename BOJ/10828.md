## 5. 최종 코드

```cpp
#include <iostream>
#include <stack>
#include <string>
using namespace std;

int main (void) {
    ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);
    stack<int> s{};
    int N = 0;
    cin >> N;
    for(int i = 0; i < N; i++) {
        string cmd = "";
        cin >> cmd;
        if (cmd == "push") {
            int foo = 0;
            cin >> foo;
            s.push(foo);
        } else if (cmd == "pop") {
            if(s.empty()) {
                cout << -1 << '\n';
            } else {
                cout << s.top() << '\n';
                s.pop();
            }
        } else if (cmd == "size") {
            cout << s.size() << '\n';
        } else if (cmd == "empty") {
            cout << (s.empty()?1:0) << '\n';
        } else if (cmd == "top") {
            if(s.empty()) {
                cout << -1 << '\n';
            } else {
                cout << s.top() << '\n';
            }
        }
    }
    return 0;
}
```
