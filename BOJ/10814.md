## 5. 최종 코드

```cpp
#include <iostream>
#include <vector>
#include <utility>
#include <string>
#include <algorithm>
using namespace std;

int main(void) {
    int N = 0;
    cin >> N;
    vector<pair<int,string>> users(N);
    for(int i = 0; i<N; i++) {
        cin >> users[i].first >> users[i].second;
    }
    stable_sort(users.begin(), users.end(), [](const pair<int,string>& a, const pair<int,string>& b){
        return a.first < b.first;
    });
    for(int i = 0; i<N; i++) {
        cout << users[i].first << ' ' << users[i].second << '\n';
    }
}
```
